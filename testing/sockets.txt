var upgrader = websocket.Upgrader{
    ReadBufferSize:  1024,
    WriteBufferSize: 1024,
}

func Upgrade(w http.ResponseWriter, r *http.Request) (*Conn, error) {
    conn, err := upgrader.upgrade(w, r, nil)
    if err != nil {
        return err
    }
    return conn
}
